func add(a: int, b: int): int {
    return a + b;
}

func sub(a: int, b: int): int {
    return a - b;
}

func mul(a: int, b: int): int {
    return a * b;
}

func div(a: int, b: int): int | nil {
    if (b == 0) {
        print("Error: division by zero");
        return nil;
    }

    return a / b;
}

func rem(a: int, b: int): int {
    return a % b;
}
